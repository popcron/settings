#region FIELD
[SerializeField]
private {PropertyType} _{PropertyName} = {DefaultValue};
#endregion

#region PROPERTY
public static {PropertyType} {PropertyName}
{
    get
    {
        return {Instance}._{PropertyName};
    }
    set
    {
        if ({Instance}._{PropertyName} != value)
        {
            {Instance}._{PropertyName} = value;
            Save();
        }
    }
}
#endregion
